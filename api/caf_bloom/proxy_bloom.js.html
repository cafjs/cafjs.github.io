'use strict';

/**
 * Proxy to a Bloom filter shared by all CAs.
 *
 * @module caf_bloom/proxy_bloom
 * @augments external:caf_components/gen_proxy
 *
 */
var caf_core = require('caf_core');
var caf_comp = caf_core.caf_components;
var genProxy = caf_comp.gen_proxy;

exports.newInstance = function($, spec, cb) {

    var that = genProxy.constructor($, spec);

    /**
     * Adds a key to the set.
     *
     * @param {string} key The string key to add.
     *
     * @memberof! module:caf_bloom/proxy_bloom#
     * @alias add
     *
     */
    that.add = function(key) {
        $._.add(key);
    };

    /**
     * Clears the filter.
     *
     * @memberof! module:caf_bloom/proxy_bloom#
     * @alias clear
     *
     */
    that.clear = function() {
        $._.clear();
    };


    /**
     * Checks that a key is in the set.
     *
     * It uses a Bloom filter that can introduce false positives, but no
     * false negatives.
     *
     * @param {string} key The key to check.
     * @param {caf.cb} cb0 A standard node.js callback that returns whether it
     * is in the set, or an error.
     *
     * @memberof! module:caf_bloom/proxy_bloom#
     * @alias has
     *
     */
    that.has = function(key, cb0) {
        $._.has(key, cb0);
    };

    Object.freeze(that);
    cb(null, that);
};
